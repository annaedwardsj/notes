Algebraic Data Type 

sealed trait Tree
object Tree {
  case object Empty                                 extends Tree
  case class  Leaf(val value: Int)                  extends Tree
  case class  Node(val left: Tree, val right: Tree) extends Tree
}
          

enum Tree {
  case Empty
  case Leaf(val value: Int)
  case Node(val left: Tree, val right: Tree)
}
